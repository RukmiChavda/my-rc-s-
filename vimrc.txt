" ------------------------------------------------------------------------------
" BASIC SETTINGS
" ------------------------------------------------------------------------------
set nocompatible
set encoding=utf-8
filetype plugin indent on
syntax on
set history=1000
set clipboard=unnamedplus

" ------------------------------------------------------------------------------
" APPEARANCE
" ------------------------------------------------------------------------------
set number
set relativenumber
set cursorline
set cursorcolumn
set showmatch
set title
set ruler
set showcmd
set laststatus=2
set cmdheight=1
set scrolloff=5
set background=dark
colorscheme desert

" ------------------------------------------------------------------------------
" INDENTATION & FORMATTING
" ------------------------------------------------------------------------------
set tabstop=4
set shiftwidth=4
set expandtab
set autoindent
set smartindent
set smarttab
set nowrap
set linebreak

" ------------------------------------------------------------------------------
" SEARCH SETTINGS
" ------------------------------------------------------------------------------
set ignorecase
set smartcase
set incsearch
set hlsearch
nnoremap <F5> :nohlsearch<CR>

" ------------------------------------------------------------------------------
" KEY MAPPINGS
" ------------------------------------------------------------------------------
nnoremap <F2> :w<CR>
nnoremap <F3> :q<CR>
nnoremap <F4> :wq<CR>
nnoremap <F6> :e!<CR>
nnoremap <F7> :set number!<CR>
nnoremap <F8> :set relativenumber!<CR>

" ------------------------------------------------------------------------------
" NAVIGATION & UI
" ------------------------------------------------------------------------------
nnoremap <Space> za
set wildmenu
set wildmode=longest:full,full
set backspace=indent,eol,start

" ------------------------------------------------------------------------------
" MOUSE & UI PERFORMANCE
" ------------------------------------------------------------------------------
set mouse=a
set ttyfast
set lazyredraw
set shortmess+=c

" ------------------------------------------------------------------------------
" BACKUP, SWAP & UNDO
" ------------------------------------------------------------------------------
set backup
set swapfile
set undofile

set backupdir=~/.vim/backup//
set directory=~/.vim/swap//
set undodir=~/.vim/undo//

" Correct directory creation (no literal '~' directory)
let s:backup_dir = expand("~/.vim/backup")
let s:swap_dir   = expand("~/.vim/swap")
let s:undo_dir   = expand("~/.vim/undo")

if !isdirectory(s:backup_dir)
  call mkdir(s:backup_dir, "p")
endif
if !isdirectory(s:swap_dir)
  call mkdir(s:swap_dir, "p")
endif
if !isdirectory(s:undo_dir)
  call mkdir(s:undo_dir, "p")
endif

" ------------------------------------------------------------------------------
" SPELL CHECK (optional)
" ------------------------------------------------------------------------------
" set spell
" set spelllang=en_us

" ------------------------------------------------------------------------------
" QUALITY OF LIFE SETTINGS
" ------------------------------------------------------------------------------
set confirm
set autoread
set hidden
set updatetime=300
set signcolumn=yes
set modeline
set visualbell
set noerrorbells

" ------------------------------------------------------------------------------
" FILETYPE-SPECIFIC SETTINGS
" ------------------------------------------------------------------------------
augroup filetype_specific
  autocmd!
  autocmd FileType python       setlocal tabstop=4 shiftwidth=4 expandtab
  autocmd FileType make         setlocal noexpandtab
  autocmd FileType tcl,sh       setlocal tabstop=4 shiftwidth=4 expandtab
  autocmd FileType verilog,vhdl setlocal tabstop=2 shiftwidth=2 expandtab
augroup END

" ------------------------------------------------------------------------------
" CUSTOM STATUS LINE (without airline)
" ------------------------------------------------------------------------------
set statusline=%f%m%r%h%w\ [%{&ff}]\ [%Y]\ [line:%l/%L,col:%c]\ [%p%%]
